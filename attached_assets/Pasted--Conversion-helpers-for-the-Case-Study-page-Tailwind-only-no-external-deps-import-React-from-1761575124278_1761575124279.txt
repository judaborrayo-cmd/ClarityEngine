import React from "react";
import { Link } from "react-router-dom";

const bookUrl = "/book-a-strategic-intro-call"; // <-- update if your site uses a different path

export function CTAButton({
  children,
  href = bookUrl,
  ariaLabel = "Book a Strategic Intro Call",
  className = "",
}: {
  children: React.ReactNode;
  href?: string;
  ariaLabel?: string;
  className?: string;
}) {
  return (
    <Link
      to={href}
      aria-label={ariaLabel}
      className={[
        "inline-flex items-center justify-center rounded-2xl px-5 py-3 font-semibold",
        "shadow-sm ring-1 ring-black/10 hover:shadow-md transition",
        "bg-black text-white hover:bg-black/90",
        className,
      ].join(" ")}
    >
      {children}
      <span className="ml-2" aria-hidden="true">→</span>
    </Link>
  );
}
