const BOOK_URL = "/book-a-strategic-intro-call"; // update if your route differs

/* ---------- Small, standalone components ---------- */
function CTAButton({ children, href = BOOK_URL, aria = "Book a Strategic Intro Call", className = "" }) {
  return (
    <a
      href={href}
      aria-label={aria}
      className={[
        "inline-flex items-center justify-center",
        "text-[15px] font-semibold rounded-2xl px-5 py-3",
        "bg-black text-white hover:bg-black/90",
        "shadow-sm ring-1 ring-black/10 hover:shadow-lg",
        "transition hover:-translate-y-[1px]",
        className,
      ].join(" ")}
    >
      {children}
      <span aria-hidden className="ml-2">→</span>
    </a>
  );
}

export function StatStrip({ items }: { items: { k: string; v: string }[] }) {

      <h2 id="ln-at-a-glance" className="sr-only">At a glance</h2>
      <div className="grid grid-cols-1 gap-3 sm:grid-cols-2 lg:grid-cols-4">
        {items.map((it, i) => (
          <div key={i} className="rounded-2xl border border-gray-200 bg-white p-4 shadow-sm">
            <div className="text-2xl font-bold">{it.k}</div>
            <div className="mt-1 text-sm text-gray-600">{it.v}</div>
          </div>
        ))}
      </div>
    </section>
  );
}
