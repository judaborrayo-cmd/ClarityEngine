import React from "react";

const bookUrl = "/book-a-strategic-intro-call"; // update if needed

export function CTAButton({
  children,
  href = bookUrl,
  ariaLabel = "Book a Strategic Intro Call",
  className = "",
}: {
  children: React.ReactNode;
  href?: string;
  ariaLabel?: string;
  className?: string;
}) {
  return (
    <a
      href={href}
      aria-label={ariaLabel}
      className={[
        "inline-flex items-center justify-center",
        "text-[15px] font-semibold rounded-2xl px-5 py-3",
        "bg-black text-white hover:bg-black/90",
        "shadow-sm ring-1 ring-black/10 hover:shadow-lg",
        "transition hover:-translate-y-[1px]",
        className,
      ].join(" ")}
    >
      {children}
      <span className="ml-2" aria-hidden="true">→</span>
    </a>
  );
}

export function StatStrip({
  items,
}: {
  items: { k: string; v: string }[];
}) {
  return (
    <section aria-labelledby="at-a-glance" className="mt-6">
      <h2 id="at-a-glance" className="sr-only">At a glance</h2>
      <div className="grid grid-cols-1 gap-3 sm:grid-cols-2 lg:grid-cols-4">
        {items.map((it) => (
          <div
            key={it.k}
            className="rounded-2xl border border-gray-200 bg-white p-4 shadow-sm"
          >
            <div className="text-2xl font-bold">{it.k}</div>
            <div className="mt-1 text-sm text-gray-600">{it.v}</div>
          </div>
        ))}
      </div>
    </section>
  );
}

function CTASection({
  title,
  body,
  dark = false,
  children,
}: {
  title: string;
  body: string;
  dark?: boolean;
  children?: React.ReactNode;
}) {
  return (
    <section
      className={[
        "my-16 rounded-3xl p-8",
        dark
          ? "bg-black text-white"
          : "bg-gradient-to-br from-gray-50 to-white border border-gray-200 text-gray-900",
      ].join(" ")}
    >
      <div className="mx-auto max-w-3xl text-center">
        <h2 className="text-xl font-semibold">{title}</h2>
        <p className={dark ? "mt-2 text-gray-200" : "mt-2 text-gray-700"}>{body}</p>
        <div className="mt-4">{children}</div>
      </div>
    </section>
  );
}

export function BeforeAfter({
  rows,
  title = "Before vs After (E-commerce System)",
}: {
  rows: { before: string; after: string }[];
  title?: string;
}) {
  return (
    <section aria-labelledby="before-after" className="my-12">
      <h2 id="before-after" className="text-lg font-semibold">{title}</h2>
      <div className="mt-4 overflow-hidden rounded-2xl border border-gray-200">
        <div className="grid grid-cols-1 md:grid-cols-2 bg-gray-50">
          <div className="p-3 text-sm font-medium text-gray-700">Before</div>
          <div className="p-3 text-sm font-medium text-gray-700 md:border-l md:border-gray-200">
            After
          </div>
        </div>
        <ul className="divide-y divide-gray-200">
          {rows.map((r, i) => (
            <li key={i} className="grid grid-cols-1 md:grid-cols-2">
              <div className="p-4 text-gray-800">{r.before}</div>
              <div className="p-4 md:border-l md:border-gray-100">{r.after}</div>
            </li>
          ))}
        </ul>
      </div>
    </section>
  );
}
