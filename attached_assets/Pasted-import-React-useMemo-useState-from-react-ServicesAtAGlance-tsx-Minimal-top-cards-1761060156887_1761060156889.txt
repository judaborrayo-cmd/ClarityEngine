 * ServicesAtAGlance.tsx
 * Minimal top cards (title + brand logos) -> detailed gradient panel below.
 * - Cards act as tabs; selected card has violet ring/glow/scale.

 * - Media supports GIF/PNG/JPG and MP4/WebM.

function Media({ src, alt = "" }) {
  const isVideo = src?.endsWith(".mp4") || src?.endsWith(".webm");
  if (isVideo) {
    return (
      <video
        className="h-full w-full rounded-xl object-cover"
        src={src}
        autoPlay
        muted
        loop
        playsInline
      />
    );
  }
  return <img className="h-full w-full rounded-xl object-cover" src={src} alt={alt} />;
}

function LogoBadge({ src, alt, label }) {
  // If you don’t have logos yet, label renders as a fallback.
  return src ? (
    <img
      src={src}
      alt={alt || label}
      className="h-6 w-auto rounded-sm bg-white/80 ring-1 ring-black/10 p-1 object-contain"
    />
  ) : (
    <span className="rounded-sm bg-white/80 px-2 py-1 text-[11px] font-medium text-gray-800 ring-1 ring-black/10">
      {label}
    </span>
  );
}

function Pill({ children }) {
  return (
    <span className="rounded-full bg-white/70 px-3 py-1 text-xs font-medium text-gray-800 ring-1 ring-black/10 backdrop-blur">
      {children}
    </span>
  );
}
