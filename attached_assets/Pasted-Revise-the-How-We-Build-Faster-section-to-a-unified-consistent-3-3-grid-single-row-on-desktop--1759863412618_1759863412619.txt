// Tooltip mini-logo
function MiniLogo({ name, img }: Mini) {
  return (
    <div className="relative group/mini">
      <img
        src={img}
        alt={name}
        className="h-4 w-4 sm:h-5 sm:w-5 object-contain opacity-85 hover:opacity-100 transition"
      />
      {/* custom tooltip */}
      <span
        className="
          pointer-events-none absolute bottom-full left-1/2 -translate-x-1/2 mb-1
          whitespace-nowrap rounded-md bg-black text-white text-[10px] px-2 py-1
          opacity-0 group-hover/mini:opacity-100 transition-opacity
        "
      >
        {name}
      </span>
    </div>
  );
}
