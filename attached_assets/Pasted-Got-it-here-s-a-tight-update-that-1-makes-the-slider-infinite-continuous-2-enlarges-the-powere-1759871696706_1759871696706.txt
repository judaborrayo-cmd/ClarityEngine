function StackChip({ name, img }: Mini) {
  return (
    <div className="inline-flex items-center gap-2 rounded-full border border-gray-200 bg-white px-3.5 py-1.5 text-[13px] leading-none shadow-sm hover:shadow transition">
      <img src={img} alt={name} className="h-[28px] w-[28px] object-contain" />
      <span className="text-gray-800">{name}</span>
    </div>
  );
}

function Card({ item }: { item: Item }) {
  return (
    <div className="shrink-0 basis-[85%] sm:basis-[48%] lg:basis-[31%] mr-4 lg:mr-6 snap-start">
      <div className="h-full rounded-2xl border border-gray-200 bg-white shadow-sm p-5 text-center">
        {/* Logo/Icon */}
        {item.main?.img ? (
          <img src={item.main.img} alt={item.name} className="h-12 w-auto mx-auto mb-2 object-contain" />
        ) : (
          <div className="h-12 w-12 mx-auto mb-2 rounded-xl bg-gray-50 border border-gray-200 flex items-center justify-center">
            <span className="text-xl">{item.main?.emoji ?? "★"}</span>
          </div>
        )}

        {/* Center title override */}
        <h3 className="font-semibold text-gray-900 text-sm sm:text-base">{item.centerLabel}</h3>
        <p className="text-[12px] text-gray-600 leading-snug mt-1">{item.desc}</p>

        {/* Powered-by chips */}
        {item.stack && item.stack.length > 0 && (
          <div className="mt-3 flex flex-wrap justify-center gap-2 max-w-[360px] mx-auto">
            {item.stack.slice(0, 4).map((m) => (
              <StackChip key={m.name} name={m.name} img={m.img} />
            ))}
            {item.stack.length > 4 && (
              <div className="inline-flex items-center gap-2 rounded-full border border-gray-200 bg-white px-3.5 py-1.5 text-[13px] leading-none shadow-sm">
                +{item.stack.length - 4}
              </div>
            )}
          </div>
        )}
      </div>
    </div>
  );
}

export default function BuildFaster() {
  const trackRef = useRef<HTMLDivElement>(null);
  const items = useMemo(() => ITEMS, []);

  // Helper: width of one card (first child) + gap
  const cardWidth = () => {
    const el = trackRef.current;
    if (!el) return 0;
    const first = el.querySelector<HTMLElement>("[data-card]");
    return first ? first.offsetWidth + 24 : el.clientWidth * 0.9;
  };

  const scrollPrev = () => {
    const el = trackRef.current;
    if (!el) return;
    // if already at start → wrap to end
    if (Math.round(el.scrollLeft) <= 0) {
      el.scrollTo({ left: el.scrollWidth - el.clientWidth, behavior: "smooth" });
      return;
    }
    el.scrollBy({ left: -cardWidth(), behavior: "smooth" });
  };

  const scrollNext = () => {
    const el = trackRef.current;
    if (!el) return;
    const atEnd = Math.ceil(el.scrollLeft + el.clientWidth) >= el.scrollWidth;
    // if at end → wrap to start
    if (atEnd) {
      el.scrollTo({ left: 0, behavior: "smooth" });
      return;
    }
    el.scrollBy({ left: cardWidth(), behavior: "smooth" });
  };

  return (
    <section className="container mx-auto px-4 py-14" aria-label="How We Build Faster">
      <div className="text-center">
        <h2 className="text-2xl sm:text-3xl font-bold mb-2">How We Build Faster</h2>
        <p className="text-gray-600 max-w-2xl mx-auto mb-6 text-sm sm:text-base">
          We're a strategic partner (fractional CMO energy) blending human insight with AI-accelerated execution — so you get strategy, design, and delivery at 10× speed.
        </p>
      </div>

      <div className="relative">
        {/* Arrows: visible, symmetric, brand-colored */}
        <button
          type="button"
          onClick={scrollPrev}
          aria-label="Previous"
          className="hidden lg:flex absolute -left-6 top-1/2 -translate-y-1/2 z-10 h-10 w-10 items-center justify-center rounded-full bg-purple-600 text-white shadow hover:bg-purple-700"
        >
          ‹
        </button>
        <button
          type="button"
          onClick={scrollNext}
          aria-label="Next"
          className="hidden lg:flex absolute -right-6 top-1/2 -translate-y-1/2 z-10 h-10 w-10 items-center justify-center rounded-full bg-purple-600 text-white shadow hover:bg-purple-700"
        >
          ›
        </button>

        {/* Track */}
        <div
          ref={trackRef}
          className="overflow-x-auto scrollbar-hide flex snap-x snap-mandatory -ml-2 pr-2"
          style={{ scrollBehavior: "smooth" }}
        >
          <div className="shrink-0 basis-[2%]" aria-hidden />
          {items.map((item) => (
            <div data-card key={item.name}>
              <Card item={item} />
            </div>
          ))}
          <div className="shrink-0 basis-[2%]" aria-hidden />
        </div>
      </div>

      <style>{`
        .scrollbar-hide::-webkit-scrollbar { display: none; }
        .scrollbar-hide { -ms-overflow-style: none; scrollbar-width: none; }
      `}</style>
    </section>
  );
}
