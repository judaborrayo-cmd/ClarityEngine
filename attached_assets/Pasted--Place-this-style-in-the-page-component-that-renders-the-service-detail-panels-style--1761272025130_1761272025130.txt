<!-- Optional helper script: show tooltip on tap for mobile -->
<script>
  document.addEventListener('click', function(e){
    const logo = e.target.closest('.ce-logo');
    if(!logo) return;
    // brief show on tap
    const tip = logo.querySelector('.ce-tip');
    if(!tip) return;
    tip.style.opacity = '1';
    tip.style.transform = 'translate(-50%, -10px)';
    setTimeout(()=>{ tip.style.opacity=''; tip.style.transform=''; }, 1200);
  }, { passive:true });
</script>
