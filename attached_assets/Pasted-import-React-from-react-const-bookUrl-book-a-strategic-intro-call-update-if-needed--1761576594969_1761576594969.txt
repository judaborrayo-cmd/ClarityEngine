import React from "react";

const bookUrl = "/book-a-strategic-intro-call"; // <-- update if needed

function CTAButton({
  children,
  href = bookUrl,
  ariaLabel = "Book a Strategic Intro Call",
  className = "",
}: {
  children: React.ReactNode;
  href?: string;
  ariaLabel?: string;
  className?: string;
}) {
  return (
    <a
      href={href}
      aria-label={ariaLabel}
      className={[
        "inline-flex items-center justify-center",
        "text-[15px] font-semibold rounded-2xl px-5 py-3",
        "bg-black text-white hover:bg-black/90",
        "shadow-sm ring-1 ring-black/10 hover:shadow-lg",
        "transition hover:-translate-y-[1px]",
        className,
      ].join(" ")}
    >
      {children}
      <span className="ml-2" aria-hidden="true">→</span>
    </a>
  );
}

/** Place directly ABOVE your existing testimonial block (do not alter the quote). */
export function TestimonialIntro() {
  return (
    <p className="mx-auto max-w-3xl text-center text-gray-700 mb-4">
      These scalable systems were first proven in a multi-industry agency environment
      and continue to guide how Clarity Engine operates today.
    </p>
  );
}

/** A light gradient wrapper for CTA sections */
function CTASection({
  title,
  body,
  dark = false,
  children,
}: {
  title: string;
  body: string;
  dark?: boolean;
  children?: React.ReactNode;
}) {
  return (
    <section
      className={[
        "my-16 rounded-3xl p-8",
        dark
          ? "bg-black text-white"
          : "bg-gradient-to-br from-gray-50 to-white border border-gray-200 text-gray-900",
      ].join(" ")}
    >
      <div className="mx-auto max-w-3xl text-center">
        <h2 className="text-xl font-semibold">{title}</h2>
        <p className={dark ? "mt-2 text-gray-200" : "mt-2 text-gray-700"}>{body}</p>
        <div className="mt-4">{children}</div>
      </div>
    </section>
  );
}

/** Mid-page CTA — add the single context line ABOVE this in your page copy. */
export function MidPageCTA() {
  return (
    <CTASection
      title="Curious what this could look like for you?"
      body="Book a Strategic Intro Call — we’ll review your current setup and outline the quickest wins."
    >
      <CTAButton>Book a Strategic Intro Call</CTAButton>
    </CTASection>
  );
}
